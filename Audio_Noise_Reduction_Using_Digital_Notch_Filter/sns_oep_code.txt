
%% FINAL NOISE REMOVAL: NOTCH FILTER CLEANUP (WITH DRAWNOW)
clear; close all; clc;

fprintf('=== FINAL NOISE REMOVAL: NOTCH FILTER CLEANUP (THREE PLOTS) ===\n\n');

%% 1. DEFINE PARAMETERS AND LOAD NOISY DATA
noisy_file = 'amir_with_noise.wav'; 
NOISE_FREQ = 6000; 
[noisy_voice, Fs] = audioread(noisy_file);
fprintf('Loaded file: %s\n', noisy_file);
fprintf('Sampling rate: %d Hz\n', Fs);

%% 2. DESIGN AND APPLY THE SINGLE NOTCH FILTER
Wo = NOISE_FREQ / (Fs/2); 
Q = 60; 
[b, a] = iirnotch(Wo, Wo/Q);
cleaned_voice = filtfilt(b, a, noisy_voice);
fprintf('Successfully applied Notch Filter (Q=%d) at %.2f Hz.\n', Q, NOISE_FREQ);

%% 3. VISUALIZATION AND COMPARISON (THREE GRAPHS)
N = min(4096, length(noisy_voice));
freqs = Fs*(0:N/2)/N;
difference_signal = noisy_voice - cleaned_voice;

% Calculate the spectrums
Y_noisy = fft(noisy_voice, N);
P_noisy = abs(Y_noisy(1:N/2+1)/N);
Y_cleaned = fft(cleaned_voice, N);
P_cleaned = abs(Y_cleaned(1:N/2+1)/N);
Y_diff = fft(difference_signal, N);
P_diff = abs(Y_diff(1:N/2+1)/N);

figure('Position', [50, 50, 1500, 400], 'Name', 'Noise Removal Analysis - Three Plots');
drawnow; % Force figure to appear instantly

% --- Plot 1: BEFORE CLEANING ---
subplot(1, 3, 1);
plot(freqs, 20*log10(P_noisy), 'b', 'LineWidth', 1.2);
title('1. BEFORE: Noisy Audio Spectrum', 'FontWeight', 'bold');
xlabel('Frequency (Hz)'); ylabel('Magnitude (dB)');
xlim([0 max(Fs/2, 10000)]); grid on;
hold on;
plot([NOISE_FREQ NOISE_FREQ], ylim, 'r--', 'LineWidth', 2, 'DisplayName', 'Noise @ 6kHz');
legend('Location', 'best');
hold off;
% --- Plot 2: AFTER CLEANING ---
subplot(1, 3, 2);
plot(freqs, 20*log10(P_cleaned), 'g', 'LineWidth', 1.2);
title('2. AFTER: Cleaned Audio Spectrum', 'FontWeight', 'bold');
xlabel('Frequency (Hz)'); ylabel('Magnitude (dB)');
xlim([0 max(Fs/2, 10000)]); grid on;
hold on;
plot([NOISE_FREQ NOISE_FREQ], ylim, 'r--', 'LineWidth', 2, 'DisplayName', 'Notch Filter');
legend('Location', 'best');
hold off;
% --- Plot 3: DIFFERENCE (REMOVED NOISE) ---
subplot(1, 3, 3);
plot(freqs, 20*log10(P_diff), 'r', 'LineWidth', 1.2);
title('3. DIFFERENCE: Removed Noise Spectrum', 'FontWeight', 'bold');
xlabel('Frequency (Hz)'); ylabel('Magnitude (dB)');
xlim([0 max(Fs/2, 10000)]); grid on;
hold on;
plot([NOISE_FREQ NOISE_FREQ], ylim, 'k--', 'LineWidth', 2, 'DisplayName', 'Noise Peak');
legend('Location', 'best');
hold off;




% Add overall title
sgtitle(sprintf('Notch Filter Noise Removal at %d Hz (Q=%d)', NOISE_FREQ, Q), 'FontSize', 14, 'FontWeight', 'bold');
%% 4. SAVE AND PLAY CLEANED AUDIO
[~, name, ext] = fileparts(noisy_file);
output_file = [name '_CLEANED_FINAL' ext];
audiowrite(output_file, cleaned_voice, Fs);

fprintf('\nSaved cleaned audio as: %s\n', output_file);

% Play comparison
fprintf('\nPlaying NOISY voice...\n');
soundsc(noisy_voice, Fs); 

audio_duration = length(noisy_voice) / Fs;
pause(audio_duration + 0.5); 

fprintf('Playing CLEANED voice...\n');
soundsc(cleaned_voice, Fs);
